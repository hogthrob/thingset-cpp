26 /* Protocol functions / categories

28 // function + data object category
29 #define TS_INFO 0x01 // read-only device information (e.g. manufacturer, device ID)
30 #define TS_CONF 0x02 // configurable settings
31 #define TS_INPUT 0x03 // input data (e.g. set-points)
32 #define TS_OUTPUT 0x04 // output data (e.g. measurement values)
33 #define TS_REC 0x05 // recorded data (history-dependent)
34 #define TS_CAL 0x06 // calibration
35 #define TS_ANY 0x09 // any of above non-exec categories 0x01-0x08
36 #define TS_EXEC 0x0B // RPC / function call
37 // function only
38 #define TS_NAME 0x0E
39 #define TS_AUTH 0x10
40 #define TS_LOG 0x11 // access log data
41 #define TS_PUB 0x12 // publication request
42 #define TS_PUBMSG 0x1F // actual publication message

76 /* Internal access rights to data objects
77  */
78 #define TS_READ_ALL (0x1U << 0) // read access for all
79 #define TS_READ_USER (0x3U << 0) // read after authentication as normal user
80 #define TS_READ_MAKER (0x7U << 0) // read after authentication as manufacturer, e.g. for factory calibration
81 #define TS_READ_MASK TS_READ_MAKER // maker has all 3 bits set
82 
83 #define TS_WRITE_ALL (0x1U << 3) // write access for all
84 #define TS_WRITE_USER (0x3U << 3) // write after authentication as normal user
85 #define TS_WRITE_MAKER (0x7U << 3) // write after authentication as manufacturer, e.g. for factory calibration
86 #define TS_WRITE_MASK TS_WRITE_MAKER
87 
88 #define TS_EXEC_ALL (0x1U << 6) // execute access for all (only for RPC)
89 #define TS_EXEC_USER (0x3U << 6) // execute after authentication as normal user
90 #define TS_EXEC_MAKER (0x7U << 6) // execute after authentication as manufacturer, e.g. for factory calibration
91 #define TS_EXEC_MASK TS_EXEC_MAKER
